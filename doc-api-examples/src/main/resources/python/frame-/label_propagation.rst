Examples
--------
.. only:: html

    .. code::

    input frame (lp.csv)
    "a"        "b"        "c"        "d"
    1,         2,         0.5,       "0.5,0.5"
    2,         3,         0.4,       "-1,-1"
    3,         1,         0.1,       "0.8,0.2"

    script

    atk.connect()()
    s = [("a", atk.int32), ("b", atk.int32), ("c", atk.float32), ("d", atk.vector(2))]
    d = "lp.csv"
    c = atk.CsvFile(d,s)
    f = atk.Frame(c)
    r = f.label_propagation("a", "b", "c", "d", "results")
    r['frame'].inspect()
    r['report']
